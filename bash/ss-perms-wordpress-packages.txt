## this helper avoids broad recursive resets of the public root ##
## each wordpress tree is divided into sftp, php, and root controlled zones ##

function ss_reset_wordpress_tree_permissions {
    local WORDPRESS_ROOT_PATH="${1}"
    local WORDPRESS_UPLOADS_SOURCE_PATH="${2}"

    if [[ ! -d "${WORDPRESS_ROOT_PATH}" ]]; then
        return 0
    fi

    ## keep the web-root shell friendly for sftp and public verification files ##
    chown "${SFTP_USER}":www-data "${WORDPRESS_ROOT_PATH}" ## must be SFTP_USER:www-data
    chmod 0775 "${WORDPRESS_ROOT_PATH}" ## must be 0775

    ## allow sftp edits for simple non-php files in the web root ##
    find "${WORDPRESS_ROOT_PATH}" -maxdepth 1 -type f ! -name "*.php" -print0 | xargs -0 --no-run-if-empty chown "${SFTP_USER}":www-data
    find "${WORDPRESS_ROOT_PATH}" -maxdepth 1 -type f ! -name "*.php" -print0 | xargs -0 --no-run-if-empty chmod 0664

    ## keep top-level wordpress php files owned by php for core updates ##
    find "${WORDPRESS_ROOT_PATH}" -maxdepth 1 -type f -name "*.php" ! -name "wp-config.php" -print0 | xargs -0 --no-run-if-empty chown www-data:www-data
    find "${WORDPRESS_ROOT_PATH}" -maxdepth 1 -type f -name "*.php" ! -name "wp-config.php" -print0 | xargs -0 --no-run-if-empty chmod 0644

    ## ensure required wp-content directories exist ##
    ss_mkdir "${WORDPRESS_ROOT_PATH}/wp-content"
    ss_mkdir "${WORDPRESS_ROOT_PATH}/wp-content/languages"
    ss_mkdir "${WORDPRESS_ROOT_PATH}/wp-content/plugins"
    ss_mkdir "${WORDPRESS_ROOT_PATH}/wp-content/temp"
    ss_mkdir "${WORDPRESS_ROOT_PATH}/wp-content/themes"
    ss_mkdir "${WORDPRESS_ROOT_PATH}/wp-content/upgrade"
    ss_mkdir "${WORDPRESS_ROOT_PATH}/wp-content/upgrade-temp-backup"

    if [[ -z "${WORDPRESS_UPLOADS_SOURCE_PATH}" ]]; then
        ss_mkdir "${WORDPRESS_ROOT_PATH}/wp-content/uploads"
    elif [[ ! -L "${WORDPRESS_ROOT_PATH}/wp-content/uploads" ]]; then
        ss_rm "${WORDPRESS_ROOT_PATH}/wp-content/uploads"
        ss_ln "${WORDPRESS_UPLOADS_SOURCE_PATH}" "${WORDPRESS_ROOT_PATH}/wp-content/uploads"
    fi

    ## allow wordpress and php to manage wp-content safely ##
    chown www-data:www-data "${WORDPRESS_ROOT_PATH}/wp-content" ## must be www-data:www-data
    chmod 0775 "${WORDPRESS_ROOT_PATH}/wp-content" ## must be 0775
    find "${WORDPRESS_ROOT_PATH}/wp-content" -maxdepth 1 -type f -print0 | xargs -0 --no-run-if-empty chown www-data:www-data
    find "${WORDPRESS_ROOT_PATH}/wp-content" -maxdepth 1 -type f -print0 | xargs -0 --no-run-if-empty chmod 0664

    ## allow wordpress updates, installs, uploads, translations, and rollback temp backups ##
    if [[ -d "${WORDPRESS_ROOT_PATH}/wp-content/languages" ]]; then
        ss_chown www-data:www-data "${WORDPRESS_ROOT_PATH}/wp-content/languages"
        find "${WORDPRESS_ROOT_PATH}/wp-content/languages" -type d -print0 | xargs -0 --no-run-if-empty chmod 0775
        find "${WORDPRESS_ROOT_PATH}/wp-content/languages" -type f -print0 | xargs -0 --no-run-if-empty chmod 0664
    fi

    if [[ -d "${WORDPRESS_ROOT_PATH}/wp-content/plugins" ]]; then
        ss_chown www-data:www-data "${WORDPRESS_ROOT_PATH}/wp-content/plugins"
        find "${WORDPRESS_ROOT_PATH}/wp-content/plugins" -type d -print0 | xargs -0 --no-run-if-empty chmod 0775
        find "${WORDPRESS_ROOT_PATH}/wp-content/plugins" -type f -print0 | xargs -0 --no-run-if-empty chmod 0664
    fi

    if [[ -d "${WORDPRESS_ROOT_PATH}/wp-content/temp" ]]; then
        ss_chown www-data:www-data "${WORDPRESS_ROOT_PATH}/wp-content/temp"
        find "${WORDPRESS_ROOT_PATH}/wp-content/temp" -type d -print0 | xargs -0 --no-run-if-empty chmod 0775
        find "${WORDPRESS_ROOT_PATH}/wp-content/temp" -type f -print0 | xargs -0 --no-run-if-empty chmod 0664
    fi

    if [[ -d "${WORDPRESS_ROOT_PATH}/wp-content/themes" ]]; then
        ss_chown www-data:www-data "${WORDPRESS_ROOT_PATH}/wp-content/themes"
        find "${WORDPRESS_ROOT_PATH}/wp-content/themes" -type d -print0 | xargs -0 --no-run-if-empty chmod 0775
        find "${WORDPRESS_ROOT_PATH}/wp-content/themes" -type f -print0 | xargs -0 --no-run-if-empty chmod 0664
    fi

    if [[ -d "${WORDPRESS_ROOT_PATH}/wp-content/upgrade" ]]; then
        ss_chown www-data:www-data "${WORDPRESS_ROOT_PATH}/wp-content/upgrade"
        find "${WORDPRESS_ROOT_PATH}/wp-content/upgrade" -type d -print0 | xargs -0 --no-run-if-empty chmod 0775
        find "${WORDPRESS_ROOT_PATH}/wp-content/upgrade" -type f -print0 | xargs -0 --no-run-if-empty chmod 0664
    fi

    if [[ -d "${WORDPRESS_ROOT_PATH}/wp-content/upgrade-temp-backup" ]]; then
        ss_chown www-data:www-data "${WORDPRESS_ROOT_PATH}/wp-content/upgrade-temp-backup"
        find "${WORDPRESS_ROOT_PATH}/wp-content/upgrade-temp-backup" -type d -print0 | xargs -0 --no-run-if-empty chmod 0775
        find "${WORDPRESS_ROOT_PATH}/wp-content/upgrade-temp-backup" -type f -print0 | xargs -0 --no-run-if-empty chmod 0664
    fi

    if [[ -z "${WORDPRESS_UPLOADS_SOURCE_PATH}" && -d "${WORDPRESS_ROOT_PATH}/wp-content/uploads" ]]; then
        ss_chown www-data:www-data "${WORDPRESS_ROOT_PATH}/wp-content/uploads"
        find "${WORDPRESS_ROOT_PATH}/wp-content/uploads" -type d -print0 | xargs -0 --no-run-if-empty chmod 0775
        find "${WORDPRESS_ROOT_PATH}/wp-content/uploads" -type f -print0 | xargs -0 --no-run-if-empty chmod 0664
    elif [[ -L "${WORDPRESS_ROOT_PATH}/wp-content/uploads" ]]; then
        chown -h root:root "${WORDPRESS_ROOT_PATH}/wp-content/uploads" ## must be root:root
    fi

    ## restrict wordpress core directories to prevent sftp edits ##
    if [[ -d "${WORDPRESS_ROOT_PATH}/wp-admin" ]]; then
        ss_chown www-data:www-data "${WORDPRESS_ROOT_PATH}/wp-admin"
        find "${WORDPRESS_ROOT_PATH}/wp-admin" -type d -print0 | xargs -0 --no-run-if-empty chmod 0755
        find "${WORDPRESS_ROOT_PATH}/wp-admin" -type f -print0 | xargs -0 --no-run-if-empty chmod 0644
    fi

    if [[ -d "${WORDPRESS_ROOT_PATH}/wp-includes" ]]; then
        ss_chown www-data:www-data "${WORDPRESS_ROOT_PATH}/wp-includes"
        find "${WORDPRESS_ROOT_PATH}/wp-includes" -type d -print0 | xargs -0 --no-run-if-empty chmod 0755
        find "${WORDPRESS_ROOT_PATH}/wp-includes" -type f -print0 | xargs -0 --no-run-if-empty chmod 0644
    fi
}

####################################################################################################
#### E. SS-Perms-WordPress-Packages: Reset WordPress Permissions (Production) ######################
####################################################################################################

## reset production wordpress permissions using explicit policy zones ##

ss_reset_wordpress_tree_permissions "/var/www/html" ""
